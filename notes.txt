drawLine(x, y, dx, dy):
a=0, b=0, diff=0
while a<dx and b<dy:
	drawPixel(x+a, y+b)
	if diff < 0: a=a+1, diff=diff+dy
	else: b=b+1, diff=diff-dx

drawCircle(x, y, r):
dy=-r.
while dy < r:
	a = x-√(r²-dy²)
	b=y+dy
	drawLine(a, b, 2*a, 0)
	r += 1
